### P1-2. `formatter.js` – `escapeHtml` 미구현

**파일**: `telegram-bot/src/utils/formatter.js`

`ai_reason` 등 Claude 응답 텍스트를 HTML 파싱 모드로 전송할 때 특수문자(`<`, `>`, `&`)가 이스케이프되지 않으면 Telegram에서 파싱 오류 또는 메시지 누락이 발생한다.

```js
// 추가할 함수
function escapeHtml(str) {
    if (!str) return '';
    return String(str)
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;');
}

// ai_reason 사용 위치에 적용
// Before: `${signal.ai_reason}`
// After:  `${escapeHtml(signal.ai_reason)}`

module.exports = { ..., escapeHtml };
```

### P1-3. `signals.js` – Rate Limiter 미구현

**파일**: `telegram-bot/src/handlers/signals.js`

분당 과다 신호 발송 방지 장치가 없어, 전략 스캔이 몰릴 경우 Telegram API 429(Too Many Requests) 오류가 발생할 수 있다.

```js
// 추가할 Rate Limiter (분당 최대 발송 수)
const MAX_SIGNALS_PER_MIN = Number(process.env.MAX_SIGNALS_PER_MIN ?? 10);
let _signalCount = 0;
let _windowStart = Date.now();

function _checkRateLimit() {
    const now = Date.now();
    if (now - _windowStart > 60_000) {
        _signalCount = 0;
        _windowStart = now;
    }
    if (_signalCount >= MAX_SIGNALS_PER_MIN) return false;
    _signalCount++;
    return true;
}

// 신호 발송 전 호출
if (!_checkRateLimit()) {
    logger.warn('[Signals] Rate limit 초과 – 신호 발송 건너뜀');
    continue;
}
```
